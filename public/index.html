  • Backend
    • Node.js Express.js TypeScript
      • Installing express
        npm i express dotenv
      • Create a server
        create .env file and add port PORT=3000
      • Installing TypeScript
        npm i typescript @types/express @types/node
      • Generating tsconfig.json
        npx tsc --init
        In tsconfig.json file: 
        {
    "compilerOptions": {
        "module": "commonjs",
        "esModuleInterop": true,
        "target": "es6",
        "moduleResolution": "node",
        "sourceMap": true,
        "outDir": "dist",
        "resolveJsonModule": true,
        "skipLibCheck": true
    },
    "lib": [
        "es2015"
    ]
}
      • Create an Express server with a .ts extension, Example: src/index.ts 
        import express, { Express, Request, Response } from "express";
import dotenv from "dotenv";

dotenv.config();

const app: Express = express();
const port = process.env.PORT || 3000;

app.get("/", (req: Request, res: Response) => {
    res.send("Express + TypeScript Server");
});

app.listen(port, () => {
    console.log(`[server]: Server is running at http://localhost:${port}`);
});
      • Running TypeScript in Node with ts-node
        npx ts-node src/index.ts
      • Watching file changes
        npm i nodemon ts-node
        Then add script in package.json
        "dev": "nodemon src/index.ts"
      • Config nodemon.json 
        {
    "watch": ["src"],
    "ext": "ts",
    "exec": "concurrently \"npx tsc --watch\" \"ts-node src/index.ts\""
}
